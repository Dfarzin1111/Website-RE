// Property card slideshow  -  auto-advances every 4.8 seconds
(function() {
  function initSlideshows() {
    document.querySelectorAll('.property-slideshow, .listing-slideshow, [data-slideshow]').forEach(function(el) {
      var slides = el.querySelectorAll('img, .slide');
      if (slides.length < 2) return;
      var dots = el.querySelectorAll('.slide-dot, .dot');
      var prevBtn = el.querySelector('.slide-prev, .prev-btn, [data-prev]');
      var nextBtn = el.querySelector('.slide-next, .next-btn, [data-next]');
      var current = 0;
      var timer;

      function goTo(n) {
        slides[current].classList.remove('active');
        slides[current].style.opacity = '0';
        if (dots[current]) dots[current].classList.remove('active');
        current = (n + slides.length) % slides.length;
        slides[current].classList.add('active');
        slides[current].style.opacity = '1';
        if (dots[current]) dots[current].classList.add('active');
      }

      function next() { goTo(current + 1); }
      function prev() { goTo(current - 1); }
      function start() { timer = setInterval(next, 4800); }
      function stop()  { clearInterval(timer); }

      // Initialize first slide
      slides.forEach(function(s, i) {
        s.style.transition = 'opacity 0.5s ease';
        s.style.position = 'absolute';
        s.style.inset = '0';
        s.style.opacity = i === 0 ? '1' : '0';
        if (i === 0) s.classList.add('active');
      });

      if (prevBtn) prevBtn.addEventListener('click', function(e){ e.stopPropagation(); stop(); prev(); start(); });
      if (nextBtn) nextBtn.addEventListener('click', function(e){ e.stopPropagation(); next(); stop(); start(); });
      dots.forEach(function(d, i) {
        d.addEventListener('click', function(e){ e.stopPropagation(); stop(); goTo(i); start(); });
      });

      el.addEventListener('mouseenter', stop);
      el.addEventListener('mouseleave', start);

      start();
    });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initSlideshows);
  } else {
    initSlideshows();
  }
})();
